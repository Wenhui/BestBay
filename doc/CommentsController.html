def create
      @item = Item.find(params[:comment][:item_id])
      @comment = @item.comments.build(params[:comment])
      @comment.user_id = current_user.id
      if @comment.save!
              @number = Comment.count
              respond_to do |format|
                      format.html {redirect_to @item}
                      format.js
              end
      end
end
